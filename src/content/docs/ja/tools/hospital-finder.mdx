import HospitalCard from '../../../../components/ui/HospitalCard.astro';
import WarningBox from '../../../../components/ui/WarningBox.astro';
import DangerBox from '../../../../components/ui/DangerBox.astro';
import GlassCard from '../../../../components/ui/GlassCard.astro';

import hospitalsRaw from '../../../../data/hospitals.json';

export const hospitals = hospitalsRaw;
export const cityOrder = ['北京', '上海', '广州', '深圳', '杭州', '南京', '武汉', '成都', '重庆', '西安', '沈阳', '香港', '台北'];
export const hospitalsByCity = cityOrder
  .map((city) => ({
    city,
    hospitals: hospitals.filter((hospital) => hospital.city === city),
  }))
  .filter(({ hospitals: cityHospitals }) => cityHospitals.length > 0);
export const cityCount = hospitalsByCity.length;

</WarningBox>
